\subsubsection{Common Test Failures}

\begin{enumerate}[leftmargin=*]
    \item \textbf{CheckoutLineInfo Parameter Issues}
    \begin{itemize}
        \item \textbf{Issue:} Tests passing \texttt{undiscounted\_unit\_price} as constructor parameter
        \item \textbf{Root Cause:} \texttt{undiscounted\_unit\_price} is a \texttt{@cached\_property}, not a constructor parameter
        \item \textbf{Solution:} Set \texttt{line.undiscounted\_unit\_price\_amount} before creating \texttt{CheckoutLineInfo}
        \item \textbf{Affected Tests:} 14+ tests in checkout base calculations
    \end{itemize}
    
    \item \textbf{Function Signature Mismatches}
    \begin{itemize}
        \item \textbf{Issue:} Incorrect number of arguments in function calls
        \item \textbf{Examples:}
        \begin{itemize}
            \item \texttt{calculate\_base\_line\_unit\_price()} takes 1 argument, tests pass 3
            \item \texttt{calculate\_base\_line\_total\_price()} takes 1-2 arguments, tests pass 3
            \item \texttt{get\_line\_total\_price\_with\_propagated\_checkout\_discount()} takes 3 arguments, tests pass 6
        \end{itemize}
        \item \textbf{Solution:} Update test calls to match actual function signatures
        \item \textbf{Affected Tests:} 20+ tests
    \end{itemize}
    
    \item \textbf{Fixture Issues}
    \begin{itemize}
        \item \textbf{Issue:} Missing or incorrect test fixtures
        \item \textbf{Example:} \texttt{fixture 'mock\_manager' not found} in account notifications tests
        \item \textbf{Solution:} Define missing fixtures or use existing ones
        \item \textbf{Affected Tests:} 4+ tests
    \end{itemize}
    
    \item \textbf{Database Constraints}
    \begin{itemize}
        \item \textbf{Issue:} Missing required fields (e.g., \texttt{shipping\_zone\_id})
        \item \textbf{Example:} \texttt{IntegrityError: null value in column "shipping\_zone\_id"}
        \item \textbf{Solution:} Create required related objects before creating test data
        \item \textbf{Affected Tests:} 8+ tests
    \end{itemize}
    
    \item \textbf{Mock Configuration}
    \begin{itemize}
        \item \textbf{Issue:} Incorrect mock setups leading to assertion failures
        \item \textbf{Example:} Mock return values don't match expected types
        \item \textbf{Solution:} Correct mock return values and types
        \item \textbf{Affected Tests:} 15+ tests
    \end{itemize}
    
    \item \textbf{DateTime Comparison Issues}
    \begin{itemize}
        \item \textbf{Issue:} Exact datetime comparisons failing due to microsecond differences
        \item \textbf{Example:} \texttt{assert datetime(2025, 12, 7, 17, 46, 5, 702981) == datetime(2025, 12, 7, 17, 46, 5, 702526)}
        \item \textbf{Solution:} Use approximate comparisons or mock time
        \item \textbf{Affected Tests:} 4+ tests
    \end{itemize}
\end{enumerate}

\section{Challenges and Solutions}

\subsection{Major Challenges}

\subsubsection{Challenge 1: Circular Import Blocking Tests}

\textbf{Problem:} Circular import blocked 334 tests (57\% of all tests) from running.

\textbf{Root Cause:} Circular dependency chain:
\begin{itemize}
    \item Migration $\rightarrow$ tasks.py $\rightarrow$ graphql/discount/utils.py $\rightarrow$
    \item graphql/product/filters/product.py $\rightarrow$ graphql/product/types/categories.py $\rightarrow$
    \item graphql/product/filters/product.py (circular)
\end{itemize}

\textbf{Solution:}
\begin{enumerate}
    \item Made imports lazy in migration file (\texttt{0067\_fulfill\_promotionrule\_variants.py})
    \item Changed product helpers to use lazy imports (\texttt{product\_helpers.py})
    \item Moved imports after class definition in categories (\texttt{categories.py})
\end{enumerate}

\textbf{Result:} All 334 tests unblocked, tests can now run successfully.

\subsubsection{Challenge 2: Low Test Coverage}

\textbf{Problem:} Coverage at 48\%, target is 80\%+.

\textbf{Root Cause:}
\begin{itemize}
    \item Critical modules with low coverage (complete\_checkout.py: 15\%, order/actions.py: 15\%)
    \item Limited test cases for complex business logic
    \item Time constraints for test development
\end{itemize}

\textbf{Solution:}
\begin{enumerate}
    \item Identified critical modules with low coverage
    \item Created comprehensive test files for each module
    \item Focused on high-impact modules first
    \item Established test infrastructure for future expansion
\end{enumerate}

\textbf{Result:} Coverage increased from 30\% to 48\%, infrastructure in place for further improvement.

\subsubsection{Challenge 3: Test Logic Failures}

\textbf{Problem:} 46+ tests failing due to incorrect test logic, fixtures, or assertions.

\textbf{Root Causes:}
\begin{itemize}
    \item Incorrect function signatures in test calls
    \item Missing required parameters in object creation
    \item Incorrect mock configurations
    \item Database constraint violations
\end{itemize}

\textbf{Solution:}
\begin{enumerate}
    \item Systematically fixing test parameter issues
    \item Correcting function signatures to match actual implementations
    \item Fixing fixture configurations
    \item Updating assertions to match actual behavior
    \item Creating required database objects before tests
\end{enumerate}

\textbf{Status:} In progress, being fixed systematically.
